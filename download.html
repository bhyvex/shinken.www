Common linux

# adduser shinken
# wget https://pypi.python.org/packages/source/S/Shinken/Shinken-2.0-RC7.tar.gz
# tar xvfz Shinken-2.0-RC7.tar.gz && cd Shinken-2.0-RC7
# python setup.py install
